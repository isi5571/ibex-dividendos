import streamlit as st
import pandas as pd
import requests
from bs4 import BeautifulSoup
from datetime import datetime

# 1. Configuración visual de la web
st.set_page_config(page_title="Próximos Dividendos IBEX 35", layout="wide")
st.title("📅 Próximos Pagos de Dividendos - IBEX 35")
st.write("Listado exclusivo de los dividendos futuros confirmados y extraídos de los comunicados oficiales de las cotizadas.")

@st.cache_data(ttl=3600) # Se actualiza automáticamente cada hora
def obtener_dividendos_reales():
    proximos_dividendos = []
    fecha_hoy = datetime.now().date()
    
    # URL del calendario de dividendos oficial para España de Investing
    url = "https://investing.com"
    
    # Cabecera para simular una visita desde un navegador real y evitar bloqueos
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
    }
    
    # Lista de control para saber qué empresas del IBEX hemos capturado
    ibex_nombres = [
        'Acciona', 'ACS', 'Aena', 'Almirall', 'Amadeus', 'ArcelorMittal', 'Bankinter', 'BBVA',
        'CaixaBank', 'Cellnex', 'Colonial', 'Enagás', 'Endesa', 'Ferrovial', 'Fluidra', 'Grifols',
        'IAG', 'Iberdrola', 'Inditex', 'Indra', 'Logista', 'Mapfre', 'Meliá Hotels', 'Merlin Prop.',
        'Naturgy', 'Puig', 'Redeia', 'Repsol', 'Rovi', 'Sabadell', 'Santander', 'Solaria',
        'Telefónica', 'Unicaja', 'Sacyr'
    ]

    try:
        # Descargamos la tabla de dividendos reales de la sesión española
        response = requests.get(url, headers=headers, timeout=10)
        soup = BeautifulSoup(response.text, 'html.parser')
        tabla = soup.find('table', {'id': 'dividendsCalendarData'})
        
        if tabla:
            filas = tabla.find('tbody').find_all('tr')
            for fila in filas:
                celdas = fila.find_all('td')
                if len(celdas) >= 4:
                    # Extraemos y limpiamos el nombre de la empresa
                    nombre_web = celdas[1].text.strip()
                    
                    # Verificamos si la empresa de la fila pertenece al IBEX 35
                    empresa_match = None
                    for b in ibex_nombres:
                        if b.lower() in nombre_web.lower():
                            empresa_match = b
                            break
                    
                    if empresa_match:
                        # Extraemos la fecha oficial de pago
                        fecha_str = celdas[0].text.strip() # Formato esperado: DD.MM.YYYY
                        try:
                            fecha_dt = datetime.strptime(fecha_str, '%d.%m.%Y').date()
                        except:
                            continue
                        
                        # FILTRO ESTRICTO: Solo guardamos si el dividendo es futuro o se paga hoy
                        if fecha_dt >= fecha_hoy:
                            # Extraemos el importe oficial bruto
                            importe_raw = celdas[2].text.strip().replace(',', '.')
                            try:
                                importe = float(importe_raw)
                                importe_formateado = f"{importe:.3f} €"
                            except:
                                importe_formateado = "Por anunciar"
                            
                            proximos_dividendos.append({
                                "Empresa": empresa_match,
                                "Próximo Dividendo (€)": importe_formateado,
                                "Fecha de Cobro": fecha_dt.strftime('%d/%m/%Y'),
                                "Estado": "🏛️ Confirmado (Hecho Relevante)",
                                "_fecha_orden": fecha_dt
                            })
    except Exception as e:
        st.error(f"Error temporal al conectar con las fuentes oficiales: {e}")

    # Si alguna empresa del IBEX no tiene un dividendo futuro anunciado a corto plazo, la rellenamos como "Sin junta convocada"
    empresas_con_dividendo = [d["Empresa"] for d in proximos_dividendos]
    for emp in ibex_nombres:
        if emp not in empresas_con_dividendo:
            proximos_dividendos.append({
                "Empresa": emp,
                "Próximo Dividendo (€)": "Por anunciar",
                "Fecha de Cobro": "Sin junta convocada",
                "Estado": "❌ Sin dividendo vigente",
                "_fecha_orden": datetime.max.date()
            })
            
    return pd.DataFrame(proximos_dividendos)

# Animación de carga de datos financieros reales
with st.spinner("Escaneando las publicaciones oficiales del Mercado Continuo..."):
    df_oficial = obtener_dividendo_reales()

st.divider()

# 2. Ordenamos automáticamente para que los pagos confirmados más cercanos aparezcan arriba
df_filtrado = df_oficial.sort_values(by="_fecha_orden", ascending=True).drop(columns=["_fecha_orden"])

# 3. Dibujar la tabla limpia en tu página web
st.dataframe(
    df_filtrado,
    column_config={
        "Empresa": st.column_config.TextColumn("Empresa", width="medium"),
        "Próximo Dividendo (€)": st.column_config.TextColumn("Importe por Acción", help="Dividendo bruto por acción extraído de la web corporativa"),
        "Fecha de Cobro": st.column_config.TextColumn("Fecha del Próximo Pago", help="Día oficial de abono en cuenta"),
        "Estado": st.column_config.TextColumn("Estado del Dividendo")
    },
    hide_index=True,
    use_container_width=True
)
